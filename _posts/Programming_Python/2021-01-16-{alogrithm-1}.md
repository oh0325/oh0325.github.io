# 프로그래머스 - 파일명 정렬


```python
from functools import cmp_to_key

def header_num(string):
    head_idx =0
    num_idx = 0
               
    l = len(string)
    down = string.lower()
    
    for i in range(l):
        if string[i].isdigit():
            head_idx =i
            break
    
    for i in range(head_idx,l):
        if (head_idx - i) >=5 :
            num_idx = i
            break
        if not (string[i].isdigit()) :
            num_idx = i
            break
    if num_idx == 0 :
        num_idx =l
       
    return down[:head_idx], int(string[head_idx : num_idx])

def compare(x,y):
    head1, num1 = header_num(x)
    head2, num2 = header_num(y)

    
    if head1 < head2:
            return -1
    elif head1 == head2:
        if num1 < num2:
                return -1
        elif num1 == num1:
                return 0
        else:
                return 1
    else:
            return 1

def solution(files):
    
        
    answer = sorted(files, key=cmp_to_key(compare))
    
           
    
    return answer
```
